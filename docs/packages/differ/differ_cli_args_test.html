Test the checkArgs function when no flag is set

 
func
 
TestArgsParsingNoFlags
(
t
 
*
testing
.
T
)
 
{

	
if
 
os
.
Getenv
(
"EXEC_NO_FLAG"
)
 
==
 
"1"
 
{

		
args
 
:=
 
types
.
CliFlags
{
}

		
checkArgs
(
&
args
)

	
}

	
cmd
 
:=
 
exec
.
Command
(
os
.
Args
[
0
]
,
 
"-test.run=TestArgsParsingNoFlags"
)

	
cmd
.
Env
 
=
 
append
(
os
.
Environ
(
)
,
 
"EXEC_NO_FLAG=1"
)

	
err
 
:=
 
cmd
.
Run
(
)

	
assert
.
NotNil
(
t
,
 
err
)

	
if
 
exitError
,
 
ok
 
:=
 
err
.
(
*
exec
.
ExitError
)
;
 
ok

.

)
)

.
Fatalf

,

)
